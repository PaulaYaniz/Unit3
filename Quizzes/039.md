from kivymd.app import MDApp


class quiz039(MDApp):
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.count = 0

    def build(self):
        return

    def setcount(self):
        self.root.ids.count_label.text = f"Count is {self.count}"

    def change(self, string: str):
        if 'sum' in string:
            self.count += 1
        self.root.ids.count_label.text = f"Count is {self.count}"


x = quiz039()
x.run()
